                            <p class="text-gray-400 mb-4">Understanding different attack vectors helps you recognize and defend against them.</p>
                            
                            <div class="grid md:grid-cols-2 gap-4 mb-4">
                                <!-- Phishing -->
                                <div class="p-5 rounded-xl bg-gradient-to-br from-red-500/10 to-transparent border border-red-500/20">

                                        <div class="w-10 h-10 rounded-lg bg-red-500/20 flex items-center justify-center">
                                            <i class="fas fa-envelope-open-text text-red-400"></i>
                                        </div>
                                        <h4 class="font-semibold text-white">Phishing</h4>
                                    </div>
                                    <p class="text-sm text-gray-400">Fraudulent emails masquerading as legitimate sources to steal credentials or install malware. Can be mass-targeted or highly personalized (spear phishing).</p>
                                </div>

                                <!-- Pretexting -->
                                <div class="p-5 rounded-xl bg-gradient-to-br from-orange-500/10 to-transparent border border-orange-500/20">
                                    <div class="flex items-center gap-3 mb-3">
                                        <div class="w-10 h-10 rounded-lg bg-orange-500/20 flex items-center justify-center">
                                            <i class="fas fa-theater-masks text-orange-400"></i>
                                        </div>
                                        <h4 class="font-semibold text-white">Pretexting</h4>
                                    </div>
                                    <p class="text-sm text-gray-400">Creating a fabricated scenario (pretext) to steal information. Attacker builds trust through a carefully crafted backstory before making requests.</p>
                                </div>

                                <!-- Baiting -->
                                <div class="p-5 rounded-xl bg-gradient-to-br from-yellow-500/10 to-transparent border border-yellow-500/20">
                                    <div class="flex items-center gap-3 mb-3">
                                        <div class="w-10 h-10 rounded-lg bg-yellow-500/20 flex items-center justify-center">
                                            <i class="fas fa-fish text-yellow-400"></i>
                                        </div>
                                        <h4 class="font-semibold text-white">Baiting</h4>
                                    </div>
                                    <p class="text-sm text-gray-400">Exploiting curiosity or greed. Examples: infected USB drives labeled "Payroll" left in parking lots, "free movie download" links containing malware.</p>
                                </div>

                                <!-- Tailgating -->
                                <div class="p-5 rounded-xl bg-gradient-to-br from-purple-500/10 to-transparent border border-purple-500/20">

                                            <i class="fas fa-door-open text-purple-400"></i>
                                        </div>
                                        <h4 class="font-semibold text-white">Tailgating / Piggybacking</h4>
                                    </div>
                                    <p class="text-sm text-gray-400">Physically following authorized personnel into restricted areas without proper credentials. Often combined with social pretext like carrying heavy items.</p>
                                </div>

    <script>
        // Initialize variables first
        const completedLessons = new Set();
        const totalLessons = document.querySelectorAll('[data-lesson]').length;

        // DOM Elements - ensure they exist before using
        const sidebar = document.getElementById('sidebar');
        const menuToggle = document.getElementById('menuToggle');
        const sidebarOverlay = document.getElementById('sidebarOverlay');
        const overallProgressEl = document.getElementById('overallProgress');
        const overallProgressBar = document.getElementById('overallProgressBar');

        // Create particles for hero section
        function createParticles() {
            const particlesContainer = document.getElementById('particles');
            if (!particlesContainer) return;
            
            for (let i = 0; i < 30; i++) {
                const particle = document.createElement('div');
                particle.className = 'particle';
                particle.style.left = Math.random() * 100 + '%';
                particle.style.top = Math.random() * 100 + '%';
                particle.style.animationDelay = Math.random() * 6 + 's';
                particle.style.animationDuration = (4 + Math.random() * 4) + 's';
                
                // Random colors
                const colors = ['#00E5FF', '#FF007F', '#FFD700'];
                particle.style.background = colors[Math.floor(Math.random() * colors.length)];
                
                particlesContainer.appendChild(particle);
            }
        }

        // Update progress
        function updateProgress() {
            const percentage = Math.round((completedLessons.size / totalLessons) * 100);
            
            if (overallProgressEl) {
                overallProgressEl.textContent = percentage + '%';
            }
            if (overallProgressBar) {
                overallProgressBar.style.width = percentage + '%';
            }

            // Update section progress bars
            const sections = ['computer-basics', 'networking', 'cybersecurity', 'kali-linux', 'blackarch', 'social-engineering'];
            sections.forEach(section => {
                const sectionLessons = document.querySelectorAll(`[data-section="${section}"]`).length || 0;
                const sectionCompleted = Array.from(completedLessons).filter(l => l.startsWith(section.split('-')[0].charAt(0))).length;
                const sectionPercentage = sectionLessons > 0 ? Math.round((sectionCompleted / 3) * 100) : 0;
                
                const progressBar = document.querySelector(`.section-progress[data-section="${section}"]`);
                if (progressBar) {
                    progressBar.style.width = sectionPercentage + '%';
                }
            });
        }

        // Sidebar toggle
        function toggleSidebar() {
            if (sidebar) {
                sidebar.classList.toggle('open');
            }
            if (sidebarOverlay) {
                sidebarOverlay.classList.toggle('hidden');
            }
        }

        // Language switcher
        function switchLanguage(lang) {
            const currentLangEl = document.getElementById('currentLang');
            if (currentLangEl) {
                currentLangEl.textContent = lang;
            }
            // In production, this would switch all text content
            console.log('Language switched to:', lang);
        }

        // Quiz functionality
        function initQuiz(quizBtn) {
            const quizId = quizBtn.dataset.quiz;
            const quizContainer = document.getElementById(quizId);
            
            if (!quizContainer) return;

            quizBtn.addEventListener('click', () => {
                quizContainer.classList.toggle('hidden');
                
                // Initialize options if not already done
                if (!quizContainer.dataset.initialized) {
                    const options = quizContainer.querySelectorAll('.quiz-option');
                    options.forEach(option => {
                        option.addEventListener('click', () => {
                            // Remove selection from siblings
                            const parent = option.closest('.quiz-options');
                            if (parent) {
                                parent.querySelectorAll('.quiz-option').forEach(opt => opt.classList.remove('selected'));
                            }
                            option.classList.add('selected');
                        });
                    });
                    quizContainer.dataset.initialized = 'true';
                }
            });

            // Check answers button
            const checkBtn = quizContainer.querySelector('.check-answers-btn');
            if (checkBtn) {
                checkBtn.addEventListener('click', () => {
                    let correct = 0;
                    const questions = quizContainer.querySelectorAll('.question');
                    
                    questions.forEach(question => {
                        const selectedOption = question.querySelector('.quiz-option.selected');
                        const options = question.querySelectorAll('.quiz-option');
                        
                        options.forEach(option => {
                            option.classList.remove('correct', 'incorrect');
                            const isCorrect = option.dataset.answer === 'true';
                            if (isCorrect) {
                                option.classList.add('correct');
                            } else if (option === selectedOption && !isCorrect) {
                                option.classList.add('incorrect');
                            }
                        });

                        if (selectedOption && selectedOption.dataset.answer === 'true') {
                            correct++;
                        }
                    });

                    // Show result
                    const resultDiv = quizContainer.querySelector('.quiz-result');
                    if (resultDiv) {
                        resultDiv.classList.remove('hidden');
                        const percentage = Math.round((correct / questions.length) * 100);
                        
                        if (percentage >= 67) {
                            resultDiv.className = 'quiz-result mt-4 p-4 rounded-lg bg-green-500/10 border border-green-500/30';
                            resultDiv.innerHTML = `<strong class="text-green-400">${correct}/${questions.length} Correct!</strong> <span class="text-gray-400">(${percentage}% - Passed)</span>`;
                        } else {
                            resultDiv.className = 'quiz-result mt-4 p-4 rounded-lg bg-red-500/10 border border-red-500/30';
                            resultDiv.innerHTML = `<strong class="text-red-400">${correct}/${questions.length} Correct</strong> <span class="text-gray-400">(${percentage}% - Try again)</span>`;
                        }
                    }
                });
            }
        }

                });
            });
        }

        // Active sidebar link on scroll
        function initScrollSpy() {
            const sections = document.querySelectorAll('section[id]');
            const navLinks = document.querySelectorAll('.sidebar-link');

            const observerOptions = {
                root: null,
                rootMargin: '-20% 0px -70% 0px',
                threshold: 0
            };

            const observer = new IntersectionObserver((entries) => {
                entries.forEach(entry => {
                    if (entry.isIntersecting) {
                        const id = entry.target.getAttribute('id');
                        navLinks.forEach(link => {
                            link.classList.remove('active');
                            if (link.getAttribute('href') === '#' + id) {
                                link.classList.add('active');
                            }
                        });
                    }
                });
            }, observerOptions);

            sections.forEach(section => observer.observe(section));
        }

        // Staggered animation on scroll
        function initScrollAnimations() {
            const observerOptions = {
                threshold: 0.1,
                rootMargin: '0px 0px -50px 0px'
            };

            const observer = new IntersectionObserver((entries) => {
                entries.forEach((entry, index) => {
                    if (entry.isIntersecting) {
                        entry.target.style.animationDelay = (index % 3) * 0.1 + 's';
                        entry.target.classList.add('stagger-item');
                        observer.unobserve(entry.target);
                    }
                });
            }, observerOptions);

            document.querySelectorAll('article').forEach(article => {
                observer.observe(article);
            });
        }

        // Initialize everything when DOM is ready

            // Initialize complete buttons
            initCompleteButtons();

            // Initialize scroll spy
            initScrollSpy();

            // Initialize scroll animations
            initScrollAnimations();

            // Initial progress update
